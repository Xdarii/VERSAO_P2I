15 def whfilter(a, weights=None, lamb=1600, p=3, ):
16  """
17  Generalized Whittaker-Handerson Graduation Method
18  Parameters
19  ----------
20  a : array-like
21  The input array, shape (n,)
22  weights : array-like or None
23  Weights
24  lamb : float
25  The relative importance between goodness of fit
26  and smoothness (smoothness increases with lamb).
27  p : integer, default 3
28  The degree of smoothness. We minimize the p-th
29  finite-differences of the graduated data. Examples:
30  p=2 Hodrick-Prescott filter;
31  p=3 Whittaker-Henderson method;
32  Note: moments 0..p-1 will be conserved by graduation
33 
34  Returns
35  -------
36  out : array
37  The smoothed data
38  References
39  ----------
40  implementation of scikits.statsmodels.tsa.filters.hp_filter.py
41  Alicja S. Nocon & William F. Scott (2012): "An extension of the
42  Whittaker-Henderson method of graduation", Scandinavian
43  Actuarial Journal, 2012:1, 70-79
44  Whittaker, E. T. (1922). "On a new method of graduation",
45  Proceedings of the Edinburgh Mathematical Society 41,63-75.
46  """
47  # input data
48  a = np.squeeze(a);
49  if a.ndim>1: raise ValueError("input array a must be 1d");
50  n = a.size;
51 
52  # weights
53  W = np.squeeze(weights) if weights is not None else np.ones(n);
54  if np.any(W==0) or not np.all(np.isfinite(W)):
55  raise ValueError("weights must be non-zero and finite.");
56  W = sparse.dia_matrix((W, 0), shape=(n,n));
57 
58  # set up difference Matrix K, shape (n-p, n)
59  # K_ij = k(j-i), l=j-i
60  # k(l) = (-1)^l Binomial(p,l) if 0<=l<=p else 0
61  l = np.arange(p+1);
62  k = (-1)**l * comb(p,l); # same as K_0j
63  diags =np.tile(k,(n,1)).T; # side-diagonal K_i,i+l; n-times k(l)
64  offsets=np.arange(p+1); # index of side-diagonals
65  K = sparse.dia_matrix((diags,offsets),shape=(n-p,n)); # K_ij
66 
67  # solve quadratic optimization problem
68  return spsolve(W+lamb*K.T.dot(K), W.dot(a));
