98 def metrique_pheno_vito(ndvi=sp.empty,seuil1=0.3,seuil2=0.75):
99  """
100  Ce script permet d'identifier les dates de debut de saison, de fin de saison à partir du NDVI à partir d'un seuil.
101  Entrée:
102  ndvi: les données de ndvi
103  seuil1: correspond au seuil autour du quel on pense trouver le debut de la vegetation
104  seuil2: correspond au seuil autour du quel on pense trouver la sénéscence
105  ndvi : correspond à l'evolution DU NDVI pour un pixel sur une année

107  Sortie:
108  une liste contenant le sos, eos, los, indMin n(indice du min),indMax (indice du max), ndvi min et ndvi max
109  """

132  test2=ndvi2<=ndviEcart2 #

136 
137  if (som1>0):

140

143 
144  Teos=sp.where( test2==1)[0][0] #
145  eos=int(sp.median(sp.where( ndvi2==ndvi2[Teos]))) + indMax +1 #

247 def metrique_pheno_param(inNdvi=sp.empty,inSos=0,inEos=0,inIndMax=0):

249  """

252

254  inNDVI: le NDVI
255  inSos: la seizaine correspondant au debut de saison
