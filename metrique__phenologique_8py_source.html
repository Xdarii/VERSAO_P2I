551  def add_action(
552  self,
553  icon_path,
554  text,
555  callback,
556  enabled_flag=True,
557  add_to_menu=True,
558  add_to_toolbar=True,
559  status_tip=None,
560  whats_this=None,
561  parent=None):
562  """Add a toolbar icon to the toolbar.
563 
564  :param icon_path: Path to the icon for this action. Can be a resource
565  path (e.g. ':/plugins/foo/bar.png') or a normal file system path.
566  :type icon_path: str
567 
568  :param text: Text that should be shown in menu items for this action.
569  :type text: str
570 
571  :param callback: Function to be called when the action is triggered.
572  :type callback: function
573 
574  :param enabled_flag: A flag indicating if the action should be enabled
575  by default. Defaults to True.
576  :type enabled_flag: bool
577 
578  :param add_to_menu: Flag indicating whether the action should also
579  be added to the menu. Defaults to True.
580  :type add_to_menu: bool
581 
582  :param add_to_toolbar: Flag indicating whether the action should also
583  be added to the toolbar. Defaults to True.
584  :type add_to_toolbar: bool
585 
586  :param status_tip: Optional text to show in a popup when mouse pointer
587  hovers over the action.
588  :type status_tip: str
589 
590  :param parent: Parent widget for the new action. Defaults None.
591  :type parent: QWidget
592 
593  :param whats_this: Optional text to show in the status bar when the
594  mouse pointer hovers over the action.
595 
596  :returns: The action that was created. Note that the action is also
597  added to self.actions list.
598  :rtype: QAction
599  """
600 
601  icon = QIcon(icon_path)
602  action = QAction(icon, text, parent)
603  action.triggered.connect(callback)
604  action.setEnabled(enabled_flag)
605 
606  if status_tip is not None:
607  action.setStatusTip(status_tip)
608 
609  if whats_this is not None:
610  action.setWhatsThis(whats_this)
611 
612  if add_to_toolbar:
613  self.toolbar.addAction(action)
614 
615  if add_to_menu:
616  self.iface.addPluginToMenu(
617  self.menu,
618  action)
619 
620  self.actions.append(action)
621 
622  return action
